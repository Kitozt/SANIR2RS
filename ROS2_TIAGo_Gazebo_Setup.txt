$ sudo apt-get update

# install Ignition Fortress

$ echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/pkgs-osrf-archive-keyring.gpg] http://packages.osrfoundation.org/gazebo/ubuntu-stable $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/gazebo-stable.list > /dev/null

$ sudo apt-get update

$ sudo apt-get install ignition-fortress

# create launch folder
$ cd ~/tiago_public_ws/src/custom_tiago_sim

vim launch/launch_tiago_in_world.py

# paste this code
from launch import LaunchDescription                            
from launch.actions import IncludeLaunchDescription             
from launch.launch_description_sources import PythonLaunchDescriptionSource                                                     
from launch.substitutions import PathJoinSubstitution           
from launch_ros.substitutions import FindPackageShare           
from launch_ros.actions import Node                                                                                             
import os     
                  
	def generate_launch_description():      
	# set gazebo model path env variable                            
	package_share_directory = FindPackageShare('tiago_gazebo').find('tiago_gazebo')                                             
    	
	# Set the environment variable using os.environ
	existing_gazebo_path = os.environ.get('GAZEBO_MODEL_PATH', '')
   	os.environ['GAZEBO_MODEL_PATH'] = existing_gazebo_path + ':' + package_share_directory + '/models'


    	# pass name of the custom world (no extension)
    	world_name = '~/worlds/test_world.sdf'

    	# Launch tiago_gazebo's main simulation launch file
    	tiago_sim = IncludeLaunchDescription(
        	PythonLaunchDescriptionSource([
            		PathJoinSubstitution([
                		FindPackageShare('tiago_gazebo'),
                		'launch',
                		'tiago_gazebo.launch.py'
            		])
        	]),
        	launch_arguments={
            		'is_public_sim': 'True',
            		'arm_type': 'no-arm',
            		'world_name': world_name  # pass world name
		}.items()
    	)

    	keyboard_teleop = Node(
            package='teleop_twist_keyboard',
            executable='teleop_twist_keyboard',
            name='keyboard_teleop',
            output='screen',
            prefix='xterm -e', # (seprate terminal launch)
            remappings=[('/cmd_vel', '/cmd_vel')]
    )

    return LaunchDescription([
        tiago_sim,
        keyboard_teleop
    ])
  
# save and exit

# if using windows subsystem for Linux (AKA WSL)
# make sure wsl is updated through window's powershell

# make sure ros2 is updated
$ sudo apt update
$ sudo apt upgrade

# install base tools and dev tools
$ sudo apt install ros-humble-ros-base
$ sudo apt install ros-dev-tools

# source ros2 everytime you open a new terminal when working in ros2
$ source directory/file
# example
$ source /opt/ros/humble/setup.bash

# you can also make it so that it automatically sources every time you open a new terminal
$ echo "source /opt/ros/humble/setup.bash" >> ~/.bashrc
# To undo this, locate your system’s shell startup script and remove the appended source command.

# how to run program
$ ros2 run <package_name> <executable_name>
# example
$ ros2 run turtlesim turtlesim_node

# command to check node list
$ ros2 node list

# add this at the end to change the name of the program's node or else it will be defaulted
$ --ros-args --remap __node:=MY_NAME
# example
$ ros2 run turtlesim turtlesim_node --ros-args --remap __node:=my_turtle

# or load a parameter files with this
$ --ros-args ==params-file FILE_NAME
# example
$ ros2 run turtlesim turtlesim_node --ros-args --params-file turtlesim.yaml

# use this to check what's connected to the nodes
$ ros2 node info /name_of_node
# example
$ ros2 node info /my_turtle

# run this for a visual display of what's connected to the nodes
$ ros2 run rqt_graph rqt_graph

# when using colcon move to the main directory you're testing/working on and run colcon build
$ colcon build --symlink-install
# use this to test the package - there are some errors you can ignore some times
$ colcon test
# then source the package, it will usually be this
$ . install/setup.bash || source install/setup.bash

# if you don't want to build a specific package place an empty file named COLCON_IGNORE inside of the directory

# packages should be created in the src directory, never in the root folder
